 CREATE TABLE "SYSMAN"."TEST" 
   ("ID" NUMBER(*,0) PRIMARY KEY, 
	"CODE" VARCHAR2(50 BYTE) NOT NULL, 
	"NUMBER" NUMBER(*,0) NOT NULL, 
	"FILENAMES" VARCHAR2(100 BYTE), 
	"ERROR" VARCHAR2(100 BYTE) )
   TABLESPACE "SYSAUX";

CREATE SEQUENCE dept_seq 
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;


CREATE OR REPLACE TRIGGER dept_bir 
BEFORE INSERT ON "SYSMAN"."TEST"
FOR EACH ROW

BEGIN
  SELECT dept_seq.NEXTVAL
  INTO   :new.ID
  FROM   dual;
END;
